\begin{frame}{Aprendizado de Máquina}
	\begin{itemize}
	\item As técnicas de \alert{Aprendizado de Máquina} têm sido aplicadas com sucesso em um grande número de problemas reais em diversos domínios
	\bigskip
	\pause
	\item Características: natureza inferencial e a boa capacidade de generalização dos métodos e técnicas desta área
	\bigskip
	\item Algoritmos capazes de aprender padrões por meio de exemplos, baseado-se em dados previamente disponíveis
	\end{itemize}
\end{frame}

\begin{frame}{Visão Computacional}
	\begin{itemize}
		\item A \alert{Visão Computacional} é uma área que procura desenvolver métodos capazes de replicar nos computadores as capacidades da visão humana.
		\pause
		\bigskip
		\item Procedimentos de extração de características de imagens envolviam um grande esforço devido ao processamento
		\bigskip
		\item Análises de especialistas eram utilizadas para descobrir as regras necessárias para o reconhecimento e extração de certos padrões
	\end{itemize}
\end{frame} 

\begin{frame}{Deep Learning}
	\begin{itemize}
	\item Algoritmos capazes de lidar com a demanda de cálculos matemáticos complexos automatizados
	\bigskip
	\item Necessidade de ferramentas e algoritmos mais sofisticados 
	\begin{itemize}
		\item Complexidade de problemas computacionais
		\item Grande volume de dados constantemente gerados
	\end{itemize}
	\bigskip
	\pause
	\item Ténicas de \alert{Deep Learning}: conjunto de camadas de redes neurais artificias 
	\begin{itemize}
		\item Simplicidade da construção das redes
		\item Escalabilidade
		\item Transferência de conhecimento
	\end{itemize}
	\end{itemize}
\end{frame}
