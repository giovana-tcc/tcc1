\begin{frame}{Métrica de Desempenho}
\begin{itemize}
	\item Métrica Erro Quadrático Médio (MSE), definido como:
	\begin{equation}
	\nonumber
	\textrm{MSE} = \frac{1}{n}\sum_{i=1}^n (y_i - \hat{y}_i)^{2} \label{eq:mse}
	\end{equation}
	\begin{itemize}
		\item $y_i$ é o valor real esperado para o exemplo $i$
		\smallskip
		\item  $\hat{y}_i$ é o valor previsto pelo modelo para o exemplo $i$
		\smallskip
		\item  $n$ é o número de amostras presentes no conjunto de dados
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Limpeza e Pré-Processamento da Base de Dados}
	\begin{itemize}
		\item Procedimento necessário para estruturação da base de dados sem que haja sobrecarga computacional
		\medskip
		\item Redimensionamento das imagens
		\begin{itemize}
			\item Tamanhos iguais de altura e largura
			\item Dimensão de $128 \times 128$ pixels
		\end{itemize}
		\medskip
		\item Selecionar imagens do espaço de cores RGB
		\begin{itemize}
			\item Eliminar imagens com outros espaços de cores
			\item Eliminar imagens em tons de cinza
		\end{itemize}
		\medskip
		\item Aproximadamente $900$ imagens foram descartadas
		\medskip
		\item Imagens convertidas para o espaço de cores CIELab
	\end{itemize}
\end{frame}
