\begin{frame}{Redes Neurais Convolucionais}
	\begin{itemize}
		\item Redes \emph{feedfoward} com muitas camadas ocultas
		\medskip
		\item Cada camada possui uma funcionalidade básica específica e produz um \alert{mapa de características}
		\medskip
		\pause
		\item Realização de operações de \alert{convolução}
		\begin{itemize}
			\item Extração de características por meio de um filtro
			\item Parâmetros: extensão espacial, \emph{stride}, \emph{zero-padding}, etc
		\end{itemize}
		\medskip
		\pause
		\item Camada \alert{\emph{pooling}} responsável por produzir um mapa de características condensado
		\medskip
		\pause
		\item Camadas \alert{completamente conectadas} baseadas nos conceitos de uma rede \emph{Multilayer Perceptron}
	\end{itemize}
\end{frame}

\begin{frame}{\emph{Transfer Learning}}
	\begin{itemize}
		\item Permite utilizar uma rede pré-treinada e adaptá-la a um novo conjunto de dados
		\bigskip
		\item Modelos treinados com um conjunto genérico de dados conseguem capturar características universais
		\bigskip
		\pause
		\item Treinamentos com conjuntos de dados bastante conhecidos
		\begin{itemize}
			\item ImageNet
			\item Places50
		\end{itemize}
		\bigskip
		\pause
		\item Modificação na camada de saída para obter o aprendizado desejado
	\end{itemize}
\end{frame}
